            function passwordreset() {

                if (!validate($("#UserName"))) {
                    return;
                }
                if (!validate($("#ccode"))) {
                    return;
                }

                var data = $("#UserName").val().toUpperCase();
                var datalength = 13;
                if (data.substring(0, 2) == 'AP') {
                    datalength = 13;
                } else {
                    datalength = 5;
                }
                if ($("#UserName").val().length < datalength) {
                    alert(" Enter Register Number");
                    return;
                } else {
                    $('#frmSL').attr('action', '/srmapstudentcorner/StudentPasswordResetInitiate');
                    $('#frmSL').submit();
                }
            }

            function validate(argObject) {
                var isValid = false;
                var regexstring = argObject.attr("patterns");
                var pattern = new RegExp(regexstring);
                isValid = pattern.test(argObject.val());
                if (!isValid) {
                    alert(argObject.attr("itemref") + " has invalid characters or Empty");
                    argObject.val("");
                }
                return isValid;
            }
